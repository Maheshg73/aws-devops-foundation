pipeline {
    agent any

    parameters {
        choice(
            name: 'ENVIRONMENT',
            choices: ['dev', 'test', 'prod'],
            description: 'Target environment'
        )
    }

    environment {
        DEMO_SECRET = credentials('demo-secret')
    }

    stages {
        stage('Checkout') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/Maheshg73/aws-devops-foundation.git'
            }
        }

        stage('Environment Validation') {
            steps {
                sh '''
                  echo "Running pipeline for environment: ${ENVIRONMENT}"

                  if [ "${ENVIRONMENT}" = "prod" ]; then
                    echo "Production safeguards enabled"
                  fi
                '''
            }
        }

        stage('Provisioning') {
            steps {
                sh '''
                  chmod +x week-01/setup.sh
                  ./week-01/setup.sh
                '''
            }
        }
    }
}
